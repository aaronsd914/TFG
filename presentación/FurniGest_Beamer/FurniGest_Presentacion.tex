\begin{frame}
  \frametitle{Contexto}
  \framesubtitle{Sector del comercio minorista de muebles}
  \begin{itemize}
    \item Predominio de PYMEs con herramientas gen\'ericas \smallskip
    \begin{itemize}
      \item Hojas de c\'alculo, papel, software de facturaci\'on b\'asico
    \end{itemize} \medskip
    \item Problemas recurrentes \smallskip
    \begin{itemize}
      \item P\'erdida de trazabilidad de pedidos \smallskip
      \item Duplicaci\'on de datos de clientes \smallskip
      \item Sin m\'etricas de negocio en tiempo real
    \end{itemize} \medskip
    \item Sistemas de Informaci\'on de Gesti\'on (MIS / ERP) \smallskip
    \begin{itemize}
      \item Capturan, almacenan, procesan y distribuyen datos de negocio
    \end{itemize}
  \end{itemize}
\end{frame}

\subsection{Motivaci\'on}
\begin{frame}
  \frametitle{Motivaci\'on}
  \framesubtitle{Limitaciones de las soluciones existentes}
  \begin{itemize}
    \item ERPs gen\'ericos del mercado \smallskip
    \begin{itemize}
      \item \textbf{Odoo, SAP Business One, Holded} \smallskip
      \item Configuraci\'on compleja, coste elevado, m\'odulos no espec\'ificos
    \end{itemize} \medskip
    \item Conceptos clave del dominio \alert{no cubiertos} \smallskip
    \begin{itemize}
      \item Albar\'an de entrega \smallskip
      \item Fianza \smallskip
      \item Rutas de reparto por cami\'on \smallskip
      \item Liquidaci\'on de transporte
    \end{itemize} \medskip
    \item \textbf{FurniGest:} aplicaci\'on a medida con vocaci\'on de \alert{producci\'on real}
  \end{itemize}
\end{frame}

\section{Objetivos}

\subsection{Objetivo general}
\begin{frame}
  \frametitle{Objetivo general}
  \begin{block}{}
    Dise\~nar, implementar y desplegar en producci\'on una aplicaci\'on web de
    gesti\'on integral para una tienda de muebles que centralice clientes,
    pedidos, log\'istica, finanzas y anal\'iticas de negocio.
  \end{block}
  \bigskip
  \begin{itemize}
    \item Adaptada al \alert{flujo real} del negocio \smallskip
    \item Desplegada en \alert{infraestructura cloud} \smallskip
    \item Verificada con \alert{quality gate} y CI/CD
  \end{itemize}
\end{frame}

\subsection{Objetivos espec\'ificos}
\begin{frame}
  \frametitle{Objetivos espec\'ificos (1/2)}
  \begin{itemize}
    \item \textbf{OBJ-01.} CRUD completo de clientes, productos y proveedores \smallskip
    \item \textbf{OBJ-02.} M\'aquina de estados del albar\'an (5 estados) \smallskip
    \item \textbf{OBJ-03.} Movimientos financieros autom\'aticos \smallskip
    \item \textbf{OBJ-04.} M\'odulo de log\'istica (camiones, manifiestos, liquidaci\'on) \smallskip
    \item \textbf{OBJ-05.} Motor de anal\'iticas (RFM, cesta, predicci\'on de Holt)
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Objetivos espec\'ificos (2/2)}
  \begin{itemize}
    \item \textbf{OBJ-06.} Asistente IA con LLM (Llama-3 v\'ia Groq) \smallskip
    \item \textbf{OBJ-07.} Cobro con tarjeta integrando Stripe Checkout \smallskip
    \item \textbf{OBJ-08.} Suite de pruebas (unitarias, integraci\'on y E2E) \smallskip
    \item \textbf{OBJ-09.} Pipeline CI/CD con GitHub Actions y SonarCloud \smallskip
    \item \textbf{OBJ-10.} Despliegue en producci\'on (Railway + Vercel)
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Metodolog\'ia de desarrollo}
  \begin{itemize}
    \item Enfoque \textbf{\'agil} con iteraciones cortas \medskip
    \item \textbf{Trunk-Based Development} \smallskip
    \begin{itemize}
      \item Rama \texttt{main} \'unica de larga vida \smallskip
      \item Ramas \texttt{feature/*} y \texttt{fix/*} de corta vida \smallskip

    \end{itemize} \medskip
    \item \textbf{Conventional Commits} \smallskip
    \begin{itemize}
      \item \texttt{feat}, \texttt{fix}, \texttt{refactor}, \texttt{test}, \texttt{docs}, \texttt{chore}
    \end{itemize} \medskip
    \item Estandar \textbf{IEEE 830} para los requisitos
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Algoritmos avanzados}
  \begin{itemize}
    \item \textbf{Segmentaci\'on RFM} de clientes \smallskip
    \begin{itemize}
      \item Recency, Frequency, Monetary $\to$ VIP, En crecimiento, En riesgo, Ocasional \smallskip
      \item Complejidad $O(n)$
    \end{itemize} \medskip
    \item \textbf{An\'alisis de cesta} (Market Basket Analysis) \smallskip
    \begin{itemize}
      \item Soporte, confianza y \emph{lift} de pares de productos \smallskip
      \item Complejidad $O(m \cdot k^2)$
    \end{itemize} \medskip
    \item \textbf{Predicci\'on con suavizado exponencial de Holt} \smallskip
    \begin{itemize}
      \item Doble suavizado (nivel + tendencia) \smallskip
      \item Intervalos de predicci\'on al 80\,\%
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{Asistente de IA}
  \begin{itemize}
    \item \textbf{Llama-3.1-8b-instant} v\'ia Groq \smallskip
    \begin{itemize}
      \item Latencia sub-segundo, API compatible OpenAI
    \end{itemize} \medskip
    \item Inyecci\'on de m\'etricas de negocio en tiempo real \smallskip
    \begin{itemize}
      \item Ventas, RFM, cesta, predicci\'on $\to$ JSON reducido ($\leq$ 120 puntos)
    \end{itemize} \medskip
    \item \emph{Prompt engineering} sin \emph{fine-tuning} \smallskip
    \begin{itemize}
      \item \texttt{temperature=0.2}, \texttt{max\_tokens=1024}
    \end{itemize} \medskip
    \item Genera \alert{respuestas y gr\'aficos} (especificaciones JSON renderizadas con Chart.js)
  \end{itemize}
\end{frame}
